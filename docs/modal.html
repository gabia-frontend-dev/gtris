Modal

To style an Modal, add the gt-modal classname.

· Set the modal by specifying the target(URL, ID or class).

<div class="gt-modal">
    <div class="gt-modal-header">모달 타이틀</div>
    <div class="gt-modal-body">모달 내용</div>
    <div class="gt-modal-close"><button data-modal="hide" title="닫기">close</button></div>
</div>

gtris.ui.modal.open({
    target: 'Set the URL, ID or Class.'
})

· Callback that fires right after completed or closed content is displayed.

gtris.ui.modal.open({
    target: 'modal/m2.html',
    completed: function() {
        alert('modal open completed');
    },
    closed: function() {
        alert('modal closed');
    }
})

· Display multiple modals sequentially on the screen.

gtris.ui.modal.open({
    target: 'modal/m3.html',
    completed: function() {
        $('#multiple').click(function(event) {
            gtris.ui.modal.open({
                target: 'modal/m1.html'
            })
        });
    }
})

· Set the overlay opacity level. Range: 0 to 1.

gtris.ui.modal.open({
    target: 'modal/m1.html',
    overlayOpacity: 1
})

· Allows the user to close the modal by clicking the overlay.

gtris.ui.modal.open({
    target: 'modal/m1.html',
    overlayColor: '#00ffaa'
})

· Allows the user to close the modal by clicking the overlay.

gtris.ui.modal.open({
    target: 'modal/m1.html',
    overlayClose: true
})

· Allows the user to close the modal when press esc.

gtris.ui.modal.open({
    target: 'modal/m1.html',
    escKey: false
})


· Set horizontal and vertical position of modal.

gtris.ui.modal.open({
    target: 'modal/m1.html',
    position: ['left', 'top']
})
